class MyQueue:
    
    def __init__(self):
        self.stack1 = []
        self.stack2 = []

    """
    @param: element: An integer
    @return: nothing
    """
    def push(self, element):
        self.stack1.append(element)

    """
    @return: An integer
    """
    def pop(self):
        while len(self.stack1) > 1:
            self.stack2.append(self.stack1.pop())
        last_item = self.stack1.pop()
        while self.stack2:
            self.stack1.append(self.stack2.pop())
        return last_item
    """
    @return: An integer
    """
    def top(self):
        while len(self.stack1) > 1:
            self.stack2.append(self.stack1.pop())
        last_item = self.stack1.pop()

        while self.stack2:
            self.stack1.append(self.stack2.pop())
        
        return last_item
